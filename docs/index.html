Quick Start

Get started with Vomee in just a few lines of code 

# Install Vomee
pip install git+https://github.com/weixijia/Vomee.git

# Basic usage
from vomee import VomeeCapture

# Initialize the platform
capture = VomeeCapture(
    video=True,
    audio=True,
    mmwave=True,
    skeleton=True
)

# Start synchronized capture
capture.start()
capture.record(duration=10)  # Record for 10 seconds
data = capture.stop()

# Save multimodal data
capture.save(data, 'output/session_001')
